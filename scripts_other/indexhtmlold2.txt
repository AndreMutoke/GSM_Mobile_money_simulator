    <script>
        // Définition des états d'affichage
        let currentMenuState = "MAIN"; // MAIN, MENU_USSD, RECOLE_NUM, RECOLE_MONTANT, RECOLE_PIN, TRANS_PIN, CONFIRMATION, SOLDE, CHARGEMENT, ERREUR, TRANS_OK
        let currentInput = ""; // Saisie clavier active
        
        let ussdString = ""; // Ex: *1234#
        let transactionOption = ""; // Option sélectionnée (Envoi, Achat...)
        let transactionNumber = ""; // Numéro destinataire
        let transactionAmount = ""; // Montant
        let transactionPIN = ""; // Code PIN saisi
        let backendMessage = ""; // Message retourné par le serveur Python

        // Options du menu principal MTK Global Money
        const menuOptions = [
            "1. Consulter mon compte",
            "2. Envoyer de l'argent",
            "3. Achat Crédit/Forfaits",
            "4. Paiement facture"
        ];
        let menuIndex = 0;

        // Fonction d'envoi de requêtes asynchrones à l'ESP32
        function envoyerCommandeESP32(action, option, numero, montant, pin = "") {
            const url = `/commande?action=${encodeURIComponent(action)}&option=${encodeURIComponent(option)}&numero=${encodeURIComponent(numero)}&montant=${encodeURIComponent(montant)}&pin=${encodeURIComponent(pin)}`;
            
            console.log("Envoi REQ -> " + url);
            
            fetch(url)
                .then(response => response.text())
                .then(data => {
                    console.log("Réponse REQ -> " + data);
                    traiterReponsePasserelle(data);
                })
                .catch(error => {
                    console.error("Erreur HTTP:", error);
                    backendMessage = "Erreur de connexion reseau ESP32";
                    currentMenuState = "ERREUR";
                    renderScreen();
                });
        }

        // Parse et applique la réponse de la passerelle
        function traiterReponsePasserelle(reponse) {
            const parts = reponse.split(';');
            const type = parts[0];
            const statut = parts[1];
            let message = parts[2] || "";

            // Convertir les séparateurs de ligne neutres "|" en sauts de ligne "\n"
            message = message.replace(/\|/g, '\n');

            if (type === "REP_SOLDE" || type === "REP_TRANS" || type === "REPONSE") {
                if (statut === "OK") {
                    backendMessage = message;
                    currentMenuState = (type === "REP_SOLDE") ? "SOLDE" : "TRANS_OK";
                } else {
                    backendMessage = message; // PIN incorrect, Solde insuffisant, Destinataire absent, etc.
                    currentMenuState = "ERREUR";
                }
            } else if (type === "REP_INFO") {
                console.log("Info locale de l'ESP32 : " + message);
                return;
            } else {
                backendMessage = reponse || "Erreur inconnue";
                currentMenuState = "ERREUR";
            }
            renderScreen();
        }

        // Rendu graphique dynamique de l'écran LCD
        function renderScreen() {
            const screen = document.getElementById('lcd-screen');
            screen.innerHTML = "";

            if (currentMenuState === "MAIN") {
                screen.innerHTML = `
                    <div class="flex justify-between items-center text-xs border-b border-[#4e5b43] pb-1">
                        <span>MTK Global RDC</span>
                        <span>12:00</span>
                    </div>
                    <div class="text-center text-[26px] my-auto py-2 tracking-widest">${currentInput || "Tapez USSD"}</div>
                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-1">
                        <span>Repertoire</span>
                        <span>${currentInput.length > 0 ? "Effacer" : "Menu"}</span>
                    </div>
                `;
            }
            else if (currentMenuState === "MENU_USSD") {
                screen.innerHTML = `
                    <div class="text-xs font-bold border-b border-[#4e5b43] pb-0.5">MTK Global Money</div>
                    <div class="flex flex-col text-[16px] leading-[18px] my-1">
                        ${menuOptions.map((opt, i) => `
                            <div class="${i === menuIndex && currentInput === '' ? 'bg-[#4e5b43] text-[#7a8e69] px-1' : ''}">${opt}</div>
                        `).join('')}
                    </div>
                    <div class="mt-auto mb-1">
                        <div class="text-[14px] bg-[#4e5b43] text-[#7a8e69] px-1 text-center font-bold tracking-widest">${currentInput || "_"}</div>
                    </div>
                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>Annuler</span>
                        <span>${currentInput.length > 0 ? "Effacer" : "Sélect."}</span>
                    </div>
                `;
            }
            else if (currentMenuState === "RECOLE_NUM") {
                screen.innerHTML = `
                    <div class="text-xs border-b border-[#4e5b43] pb-0.5">${transactionOption}</div>
                    <div class="my-auto">
                        <div class="text-[14px]">Numéro destinataire :</div>
                        <div class="bg-[#4e5b43] text-[#7a8e69] px-1 text-center font-bold tracking-wider mt-1">${currentInput || "_"}</div>
                    </div>
                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>Retour</span>
                        <span>${currentInput.length > 0 ? "Effacer" : "Continuer"}</span>
                    </div>
                `;
            }
            else if (currentMenuState === "RECOLE_MONTANT") {
                screen.innerHTML = `
                    <div class="text-xs border-b border-[#4e5b43] pb-0.5">Num: ${transactionNumber}</div>
                    <div class="my-auto">
                        <div class="text-[14px]">Montant en USD :</div>
                        <div class="bg-[#4e5b43] text-[#7a8e69] px-1 text-center font-bold mt-1">${currentInput || "_"} USD</div>
                    </div>
                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>Retour</span>
                        <span>${currentInput.length > 0 ? "Effacer" : "Continuer"}</span>
                    </div>
                `;
            }
            else if (currentMenuState === "RECOLE_PIN") {
                screen.innerHTML = `
                    <div class="text-xs border-b border-[#4e5b43] pb-0.5">Consultation Compte</div>
                    <div class="my-auto">
                        <div class="text-[14px]">Entrez votre code PIN:</div>
                        <div class="bg-[#4e5b43] text-[#7a8e69] px-1 text-center font-bold tracking-widest mt-1">
                            ${"*".repeat(currentInput.length) || "_"}
                        </div>
                    </div>
                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>Retour</span>
                        <span>${currentInput.length > 0 ? "Effacer" : "Consulter"}</span>
                    </div>
                `;
            }
            else if (currentMenuState === "TRANS_PIN") {
                screen.innerHTML = `
                    <div class="text-xs border-b border-[#4e5b43] pb-0.5">Envoi: ${transactionAmount} USD</div>
                    <div class="my-auto">
                        <div class="text-[14px]">Entrez code PIN:</div>
                        <div class="bg-[#4e5b43] text-[#7a8e69] px-1 text-center font-bold tracking-widest mt-1">
                            ${"*".repeat(currentInput.length) || "_"}
                        </div>
                    </div>
                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>Annuler</span>
                        <span>${currentInput.length > 0 ? "Effacer" : "Confirmer"}</span>
                    </div>
                `;
            }
            else if (currentMenuState === "CHARGEMENT") {
                screen.innerHTML = `
                    <div class="text-center my-auto flex flex-col items-center justify-center py-6">
                        <div class="animate-spin rounded-full h-8 w-8 border-b-2 border-[#1d2319] mb-2"></div>
                        <div class="text-[18px] tracking-wide">Communication...</div>
                    </div>
                `;
            }
            else if (currentMenuState === "SOLDE") {
                screen.innerHTML = `
                    <div class="text-xs font-bold border-b border-[#4e5b43] pb-0.5">Compte MTK Global</div>
                    <div class="text-[16px] my-auto text-center font-semibold leading-none py-1 whitespace-pre-line tracking-wide">
                        ${backendMessage}
                    </div>
                    <div class="flex justify-center text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>[0] Quitter</span>
                    </div>
                `;
            }
            else if (currentMenuState === "TRANS_OK") {
                screen.innerHTML = `
                    <div class="text-xs font-bold border-b border-[#4e5b43] pb-0.5">MTK Global Money</div>
                    <div class="text-[14px] my-auto text-center leading-tight whitespace-pre-line text-green-900 font-semibold py-1">
                        ${backendMessage}
                    </div>
                    <div class="flex justify-center text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>[0] Menu Principal</span>
                    </div>
                `;
            }
            else if (currentMenuState === "ERREUR") {
                screen.innerHTML = `
                    <div class="text-xs font-bold border-b border-[#4e5b43] pb-0.5 text-red-800">Erreur USSD</div>
                    <div class="text-[14px] my-auto text-center leading-relaxed font-bold">
                        ${backendMessage || "Operation invalide."}
                    </div>
                    <div class="flex justify-center text-xs border-t border-[#4e5b43] pt-0.5">
                        <span>[0] Recommencer</span>
                    </div>
                `;
            }
        }

        /* --- Gestion des touches Clavier --- */
        function handleKey(key) {
            if (currentMenuState === "MAIN") {
                currentInput += key;
            } 
            else if (currentMenuState === "MENU_USSD") {
                currentInput += key;
            }
            else if (currentMenuState === "RECOLE_NUM" || currentMenuState === "RECOLE_MONTANT") {
                currentInput += key;
            }
            else if (currentMenuState === "RECOLE_PIN" || currentMenuState === "TRANS_PIN") {
                if (currentInput.length < 4 && !isNaN(key)) {
                    currentInput += key;
                }
            }
            else if (["SOLDE", "ERREUR", "TRANS_OK"].includes(currentMenuState)) {
                if (key === "0") {
                    currentInput = "";
                    currentMenuState = "MAIN";
                }
            }
            renderScreen();
        }

        function handleClear() {
            if (currentInput.length > 0) {
                currentInput = currentInput.slice(0, -1);
                renderScreen();
            }
        }

        // Navigation Flèches
        function handleArrowUp() {
            if (currentMenuState === "MENU_USSD" && menuIndex > 0) {
                menuIndex--;
                renderScreen();
            }
        }

        function handleArrowDown() {
            if (currentMenuState === "MENU_USSD" && menuIndex < menuOptions.length - 1) {
                menuIndex++;
                renderScreen();
            }
        }

        // Touche de Validation GAUCHE / Entrée
        function handleSoftLeft() {
            if (currentMenuState === "MENU_USSD") {
                let choix = currentInput !== "" ? currentInput : (menuIndex + 1).toString();
                
                if (choix === "1") {
                    currentInput = "";
                    currentMenuState = "RECOLE_PIN";
                } else if (choix === "2") {
                    transactionOption = "Envoyer argent";
                    currentInput = "";
                    currentMenuState = "RECOLE_NUM";
                } else {
                    backendMessage = "Option non supportee";
                    currentMenuState = "ERREUR";
                }
            } 
            else if (currentMenuState === "RECOLE_NUM") {
                if (currentInput.length >= 8) {
                    transactionNumber = currentInput;
                    currentInput = "";
                    currentMenuState = "RECOLE_MONTANT";
                }
            }
            else if (currentMenuState === "RECOLE_MONTANT") {
                if (currentInput.length > 0 && parseFloat(currentInput) > 0) {
                    transactionAmount = currentInput;
                    currentInput = "";
                    currentMenuState = "TRANS_PIN";
                }
            }
            else if (currentMenuState === "RECOLE_PIN") {
                if (currentInput.length === 4) {
                    transactionPIN = currentInput;
                    currentMenuState = "CHARGEMENT";
                    renderScreen();
                    setTimeout(() => {
                        envoyerCommandeESP32("SOLDE", "Consultation", "Aucun", "0", transactionPIN);
                    }, 800);
                }
            }
            else if (currentMenuState === "TRANS_PIN") {
                if (currentInput.length === 4) {
                    transactionPIN = currentInput;
                    currentMenuState = "CHARGEMENT";
                    renderScreen();
                    setTimeout(() => {
                        envoyerCommandeESP32("CONFIRME", transactionOption, transactionNumber, transactionAmount, transactionPIN);
                    }, 800);
                }
            }
            renderScreen();
        }

        // Touche de Navigation DROITE
        function handleSoftRight() {
            if (currentInput.length > 0) {
                handleClear();
                return;
            }

            if (currentMenuState === "MENU_USSD") {
                currentMenuState = "MAIN";
                currentInput = "";
            } 
            else if (currentMenuState === "RECOLE_NUM") {
                currentMenuState = "MENU_USSD";
                currentInput = "";
            }
            else if (currentMenuState === "RECOLE_MONTANT") {
                currentMenuState = "RECOLE_NUM";
                currentInput = transactionNumber;
            }
            else if (currentMenuState === "RECOLE_PIN") {
                currentMenuState = "MENU_USSD";
                currentInput = "";
            }
            else if (currentMenuState === "TRANS_PIN") {
                currentMenuState = "RECOLE_MONTANT";
                currentInput = transactionAmount;
            }
            renderScreen();
        }

        // Touche VERTE (Appel)
        function handleCall() {
            if (currentMenuState === "MAIN") {
                if (currentInput === "*1234#") {
                    currentMenuState = "CHARGEMENT";
                    renderScreen();
                    
                    setTimeout(() => {
                        envoyerCommandeESP32("APPEL_USSD", "*1234#", "Aucun", "0");
                        currentMenuState = "MENU_USSD";
                        menuIndex = 0;
                        renderScreen();
                    }, 1000);
                } else if (currentInput.length > 0) {
                    backendMessage = "Code MMI non valide.";
                    currentMenuState = "ERREUR";
                    renderScreen();
                }
            } else {
                handleSoftLeft();
            }
        }

        // Touche ROUGE (Fin)
        function handleEnd() {
            currentInput = "";
            currentMenuState = "MAIN";
            renderScreen();
        }

        // Initialisation de l'écran au démarrage
        renderScreen();

        // Support clavier physique pour une meilleure immersion
        document.addEventListener('keydown', (e) => {
            if (e.key >= '0' && e.key <= '9' || e.key === '*' || e.key === '#') {
                handleKey(e.key);
            } else if (e.key === 'Backspace') {
                handleClear();
            } else if (e.key === 'Enter') {
                handleCall();
            } else if (e.key === 'Escape') {
                handleEnd();
            } else if (e.key === 'ArrowUp') {
                handleArrowUp();
            } else if (e.key === 'ArrowDown') {
                handleArrowDown();
            }
        });

    </script>
