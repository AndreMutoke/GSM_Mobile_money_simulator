    <script>
        // --- Variables d'état ---
        let currentInput = "*1234#";
        let ussdState = "TYPING"; // États: TYPING, LOADING, MENU, MESSAGE
        let currentMenuState = "IDLE";
        
        // Variables récoltées de la transaction
        let transactionOption = "Non specifiee";
        let transactionNumber = "Aucun";
        let transactionAmount = "0 FC";

        const screenContent = document.getElementById("screen-content");
        const softLeft = document.getElementById("soft-left");
        const softRight = document.getElementById("soft-right");

        // --- COMMUNICATION ESP32 ---
        // Fonction enrichie pour envoyer les informations structurées à l'ESP32
        function envoyerCommandeESP32(commande, option = "N/A", numero = "N/A", montant = "N/A") {
            try {
                if (window.location.protocol === 'file:' || window.location.protocol === 'blob:' || window.location.protocol === 'data:') {
                    console.log("[MODE SIMULATEUR] UART Virtuel :", {
                        commande: commande,
                        option: option,
                        numero: numero,
                        montant: montant
                    });
                    return;
                }

                // Construction de l'URL avec les paramètres récoltés pour l'UART de l'ESP32
                const url = `/ussd?cmd=${encodeURIComponent(commande)}` +
                            `&op=${encodeURIComponent(option)}` +
                            `&num=${encodeURIComponent(numero)}` +
                            `&amt=${encodeURIComponent(montant)}`;

                fetch(url)
                    .then(response => console.log("Infos transmises à l'ESP32"))
                    .catch(err => console.warn("Erreur réseau (normal si hors-ligne)", err));
            } catch (erreur) {
                console.warn("[MODE HORS-LIGNE] Erreur :", erreur.message);
            }
        }

        // --- Mise à jour de l'horloge ---
        setInterval(() => {
            const now = new Date();
            document.getElementById('clock').innerText = 
                String(now.getHours()).padStart(2, '0') + ":" + 
                String(now.getMinutes()).padStart(2, '0');
        }, 1000);

        // --- Moteur de rendu de l'écran LCD ---
        function renderScreen() {
            if (ussdState === "TYPING") {
                screenContent.innerHTML = `
                    <div class="text-[1rem] tracking-widest uppercase mb-2">Composition</div>
                    <div class="text-2xl tracking-widest font-bold">${currentInput}<span class="cursor-blink">|</span></div>
                    <div class="w-[80%] h-[2px] bg-black my-2"></div>
                    <div class="text-[1rem]">APPEL pour lancer</div>
                `;
                softLeft.innerText = "Effacer";
                softRight.innerText = "Appel";
            } 
            else if (ussdState === "LOADING") {
                screenContent.innerHTML = `
                    <div class="flex-grow flex items-center justify-center">
                        Connexion en cours...
                    </div>
                `;
                softLeft.innerText = "";
                softRight.innerText = "";
            }
            else if (ussdState === "MENU" || ussdState === "MESSAGE") {
                let html = `<div class="w-full text-left font-bold text-[1.1rem] leading-none mb-1">`;
                
                if (currentMenuState === "MAIN") {
                    html += `Menu MTK Global Money<br>1.Transfert<br>2.Achat Unites<br>3.Retrait<br>4.Mon Compte<br><br>> ${currentInput}<span class="cursor-blink">_</span>`;
                    softLeft.innerText = "Annuler";
                    softRight.innerText = "Reponse";
                } 
                else if (currentMenuState === "TRANSFERT") {
                    html += `Transfert:<br>1.Vers MTK Global<br>2.Vers autre<br>0.Retour<br><br>> ${currentInput}<span class="cursor-blink">_</span>`;
                    softLeft.innerText = "Annuler";
                    softRight.innerText = "Reponse";
                }
                else if (currentMenuState === "NUM_INPUT") {
                    html += `Entrez le numéro du bénéficiaire:<br><br>> ${currentInput}<span class="cursor-blink">_</span>`;
                    softLeft.innerText = "Annuler";
                    softRight.innerText = "OK";
                }
                else if (currentMenuState === "AMT_INPUT") {
                    html += `Entrez le montant en FC:<br><br>> ${currentInput}<span class="cursor-blink">_</span>`;
                    softLeft.innerText = "Annuler";
                    softRight.innerText = "OK";
                }
                else if (currentMenuState === "CONFIRM") {
                    html += `Confirmer transfert ?<br>Dest: ${transactionNumber}<br>Montant: ${transactionAmount}<br>1. Confirmer<br>2. Annuler<br><br>> ${currentInput}<span class="cursor-blink">_</span>`;
                    softLeft.innerText = "Annuler";
                    softRight.innerText = "OK";
                }
                else if (currentMenuState === "COMPTE") {
                    html += `Mon Compte:<br>1.Solde<br>2.Changer PIN<br>0.Retour<br><br>> ${currentInput}<span class="cursor-blink">_</span>`;
                    softLeft.innerText = "Annuler";
                    softRight.innerText = "Reponse";
                }
                else if (currentMenuState === "SOLDE") {
                    html += `Votre solde est de 14500 FC.<br>Frais: 0 FC.`;
                    softLeft.innerText = "Quitter";
                    softRight.innerText = "OK";
                    ussdState = "MESSAGE";
                }
                else if (currentMenuState === "SUCCESS") {
                    html += `Succès !<br>${transactionAmount} envoyé au ${transactionNumber}.`;
                    softLeft.innerText = "Quitter";
                    softRight.innerText = "OK";
                    ussdState = "MESSAGE";
                }
                else if (currentMenuState === "ERROR") {
                    html += `Problème de connexion ou code non valide.`;
                    softLeft.innerText = "Quitter";
                    softRight.innerText = "OK";
                    ussdState = "MESSAGE";
                }

                html += `</div>`;
                screenContent.innerHTML = html;
            }
        }

        // --- Gestionnaires d'événements matériels ---
        function handleKey(char) {
            if (ussdState === "TYPING" || ussdState === "MENU") {
                if (currentInput.length < 15) {
                    currentInput += char;
                    renderScreen();
                }
            }
        }

        // Action de suppression (Clear)
        function handleClear() {
            if (ussdState === "TYPING" || ussdState === "MENU") {
                currentInput = currentInput.slice(0, -1);
                renderScreen();
            }
        }

        // Touche d'action gauche (Soft Left)
        function handleSoftLeft() {
            if (ussdState === "TYPING") {
                handleClear();
            } else if (ussdState === "MENU" || ussdState === "MESSAGE") {
                handleEnd(); // Annuler / Quitter
            }
        }

        // Touche d'action droite (Soft Right)
        function handleSoftRight() {
            if (ussdState === "TYPING") {
                handleCall();
            } else if (ussdState === "MENU" || ussdState === "MESSAGE") {
                processMenuInput();
            }
        }

        // Touche centrale OK
        function handleOK() {
            handleSoftRight();
        }

        // Fin / Réinitialisation (Touche Rouge)
        function handleEnd() {
            ussdState = "TYPING";
            currentInput = "";
            currentMenuState = "IDLE";
            transactionOption = "Non specifiee";
            transactionNumber = "Aucun";
            transactionAmount = "0 FC";
            renderScreen();
        }

        // Action de lancement d'appel (Touche Verte)
        function handleCall() {
            if (ussdState === "TYPING") {
                if (currentInput === "") return;
                
                envoyerCommandeESP32(currentInput, "Initialisation", "Aucun", "0 FC");
                
                ussdState = "LOADING";
                renderScreen();

                setTimeout(() => {
                    if (currentInput === "*1234#") {
                        ussdState = "MENU";
                        currentMenuState = "MAIN";
                        currentInput = "";
                    } else {
                        ussdState = "MESSAGE";
                        currentMenuState = "ERROR";
                        currentInput = "";
                    }
                    renderScreen();
                }, 800);
            } else if (ussdState === "MENU") {
                processMenuInput();
            }
        }

        // --- Logique enrichie du flux USSD MTK Global Money ---
        function processMenuInput() {
            if (ussdState === "MESSAGE") {
                handleEnd();
                return;
            }

            const input = currentInput;
            
            // Envoyer d'abord la commande brute à l'ESP32 pour l'interaction UART en temps réel
            envoyerCommandeESP32(input, transactionOption, transactionNumber, transactionAmount);
            
            ussdState = "LOADING";
            renderScreen();

            setTimeout(() => {
                ussdState = "MENU";
                
                if (currentMenuState === "MAIN") {
                    if (input === "1") {
                        currentMenuState = "TRANSFERT";
                        transactionOption = "Menu Transfert";
                    } else if (input === "4") {
                        currentMenuState = "COMPTE";
                        transactionOption = "Menu Compte";
                    } else {
                        currentMenuState = "ERROR";
                    }
                } 
                else if (currentMenuState === "TRANSFERT") {
                    if (input === "0") {
                        currentMenuState = "MAIN";
                        transactionOption = "MTK Global Money";
                    } else if (input === "1") {
                        currentMenuState = "NUM_INPUT";
                        transactionOption = "Transfert vers MTK Global";
                    } else if (input === "2") {
                        currentMenuState = "NUM_INPUT";
                        transactionOption = "Transfert vers autre reseau";
                    } else {
                        currentMenuState = "ERROR";
                    }
                }
                else if (currentMenuState === "NUM_INPUT") {
                    if (input.length >= 8) { // validation simple du numero
                        transactionNumber = input;
                        currentMenuState = "AMT_INPUT";
                    } else {
                        currentMenuState = "ERROR";
                    }
                }
                else if (currentMenuState === "AMT_INPUT") {
                    if (input.length > 0 && !isNaN(input)) {
                        transactionAmount = input + " FC";
                        currentMenuState = "CONFIRM";
                    } else {
                        currentMenuState = "ERROR";
                    }
                }
                else if (currentMenuState === "CONFIRM") {
                    if (input === "1") {
                        currentMenuState = "SUCCESS";
                        // Envoi final à l'ESP32 pour confirmer et valider la transaction complète !
                        envoyerCommandeESP32("CONFIRME", transactionOption, transactionNumber, transactionAmount);
                    } else {
                        currentMenuState = "MAIN";
                    }
                }
                else if (currentMenuState === "COMPTE") {
                    if (input === "0") {
                        currentMenuState = "MAIN";
                    } else if (input === "1") {
                        currentMenuState = "SOLDE";
                        transactionOption = "Consultation Solde";
                        envoyerCommandeESP32("SOLDE", transactionOption, "Aucun", "0 FC");
                    } else {
                        currentMenuState = "ERROR";
                    }
                }

                currentInput = "";
                renderScreen();
            }, 500);
        }

        // Initialisation de l'écran au démarrage
        renderScreen();

        // Support clavier physique
        document.addEventListener('keydown', (e) => {
            if (e.key >= '0' && e.key <= '9' || e.key === '*' || e.key === '#') {
                handleKey(e.key);
            } else if (e.key === 'Backspace') {
                handleClear();
            } else if (e.key === 'Enter') {
                handleCall();
            } else if (e.key === 'Escape') {
                handleEnd();
            }
        });

    </script>
