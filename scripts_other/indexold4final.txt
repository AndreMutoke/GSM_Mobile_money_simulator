        function getLieu(lat, lon, mode) {
            if (mode === 1) {

        function mettreAJourPositionEtSignaux() {
            let trajet = simulationMode === 1 ? trajetSim1 : trajetSim2;
            let btsPositions = simulationMode === 1 ? btsPositionsSim1 : btsPositionsSim2;

            currentMSLat = trajet.startLat + (trajet.endLat - trajet.startLat) * simProgress;
            currentMSLon = trajet.startLon + (trajet.endLon - trajet.startLon) * simProgress;

            let rawBts = btsPositions.map(bts => {
                let dx = bts.lon - currentMSLon;
                let dy = bts.lat - currentMSLat;
                let distance = Math.sqrt(dx*dx + dy*dy);
                let rxLev = Math.round(-50 - (distance * 450)); 
                if (rxLev < -99) rxLev = -99;
                if (rxLev > -45) rxLev = -45;
                return { id: bts.id, cellId: bts.cellId, name: bts.name, rxLev: rxLev, lat: bts.lat, lon: bts.lon };
            });

            rawBts.sort((a, b) => b.rxLev - a.rxLev);
            let accuracy = Math.floor(Math.abs(rawBts[2].rxLev) * 2.5) + Math.floor(Math.random() * 50);

            let locData = {
                mcc: 630, mnc: 89, lac: 4001,
                lat: currentMSLat.toFixed(5), lon: currentMSLon.toFixed(5),
                precision: accuracy, lieu: getLieu(currentMSLat, currentMSLon, simulationMode),
                bts: rawBts
            };

            backendMessage = JSON.stringify(locData);

            // --- ENVOI DES DONNÉES BTS AU HLR POUR SUIVI TEMPS RÉEL (Handover) ---
            updateTickCounter++;
            if (updateTickCounter % 3 === 0) { // Toutes les 3 secondes pour ne pas inonder l'UART

                fetch(`/commande?action=LOC_UPDATE&bts=${encodeURIComponent(btsStr)}`).catch(() => {});

            simProgress += 0.02; 
            if (simProgress > 1.0) simProgress = 0.0; 

            renderScreen();
        }

        function envoyerCommandeESP32(action, option, numero, montant, pin = "") {
            let url = "";
            // Si c'est une requête de localisation initiale (Scan)
            if (action === "LOC") {
                url = `/commande?action=LOC&bts=${encodeURIComponent(option)}`;
            } else {
                url = `/commande?action=${encodeURIComponent(action)}&option=${encodeURIComponent(option)}&numero=${encodeURIComponent(numero)}&montant=${encodeURIComponent(montant)}&pin=${encodeURIComponent(pin)}`;
            }
            
            if (window.location.hostname !== "192.168.4.1" && window.location.hostname !== "localhost") {
                setTimeout(() => {
                    let simResponse = "";
                    if (action === "APPEL_USSD") { simResponse = "REPONSE;OK;Menu USSD ouvert"; } 
                    else if (action === "SOLDE") {
                        if (pin === "1234") simResponse = "REP_SOLDE;OK;Solde: 320.00 USD";
                        else simResponse = "ERREUR;COMPTE_BLOQUE;PIN incorrect. COMPTE_BLOQUE";
                    } 
                    else if (action === "CONFIRME") {
                        if (option.includes("RECOVERY_")) simResponse = "REP_TRANS;OK;KYC Verifie.\\nCompte debloque !\\nNouveau PIN: 1234";
                        else {
                            if (pin === "1234") simResponse = `REP_TRANS;OK;Transfert de ${montant} USD au ${numero} reussi.`;
                            else simResponse = "ERREUR;COMPTE_BLOQUE;Code PIN incorrect. COMPTE_BLOQUE";
                        }
                    }
                    else if (action === "LOC") {
                        simResponse = "REP_LOC;OK;Validation HLR Reussie"; // Simulation validation HLR
                    }
                    traiterReponsePasserelle(simResponse);
                }, 1000);
                return;
            }

            fetch(url).then(r => r.text()).then(data => traiterReponsePasserelle(data)).catch(e => {
                backendMessage = "Erreur de connexion reseau ESP32";
                currentMenuState = "ERREUR";
                renderScreen();
            });
        }

        function traiterReponsePasserelle(reponse) {
            const parts = reponse.split(';');
            const type = parts[0];
            const statut = parts[1];
            let message = parts[2] || "";
            message = message.replace(/\\n/g, '\n').replace(/\|/g, '\n');

            if (statut === "ERREUR" && message.includes("COMPTE_BLOQUE")) {
                currentMenuState = "REC_NOM";
                currentInput = "";
                renderScreen();
                return;
            }

            // GESTION DE LA VALIDATION HLR POUR LA LOCALISATION
            if (type === "REP_LOC") {
                if (statut === "OK") {
                    currentMenuState = "APP_LOC_RESULT";
                    locViewIndex = 2; // Commencer par la carte radar graphique
                    btsListScroll = 0;
                    demarrerLoopSimulation();
                } else {
                    backendMessage = "Rejet HLR: BTS Invalides/Inconnues";
                    currentMenuState = "ERREUR";
                }
            }
            else if (type === "REP_SOLDE" || type === "REP_TRANS" || type === "REPONSE") {
                if (statut === "OK") {
                    backendMessage = message;
                    if (type === "REPONSE" && message.includes("Menu USSD ouvert")) {
                        currentMenuState = "MENU_USSD";
                        menuIndex = 0;
                        currentInput = "";
                    } else {
                        currentMenuState = (type === "REP_SOLDE") ? "SOLDE" : "TRANS_OK";
                    }
                } else {
                    backendMessage = message;
                    currentMenuState = "ERREUR";
                }
            } else if (type === "REP_INFO") {
                return;
            } else {
                backendMessage = reponse || "Erreur inconnue";
                currentMenuState = "ERREUR";
            }
            renderScreen();
        }

        function generateRadarMapHTML(locData) {
            if (!locData || !locData.bts || locData.bts.length < 3) return `<div class="text-center mt-4">Erreur carte</div>`;

            const msLat = parseFloat(locData.lat);
            const msLon = parseFloat(locData.lon);
            const scale = 550; 

            let mapHTML = `<div id="map-radar" class="radar-map relative">`;
            mapHTML += `<div class="radar-crosshair" style="top: 50%; left: 50%;">🚗</div>`;

            locData.bts.forEach((bts) => {
                let dLat = (bts.lat - msLat) * scale;
                let dLon = (bts.lon - msLon) * scale;
                
                let topPos = 50 - dLat;
                let leftPos = 50 + dLon;

                topPos = Math.max(8, Math.min(85, topPos));
                leftPos = Math.max(8, Math.min(92, leftPos));

                const isServingBTS = (bts.cellId === locData.bts[0].cellId);
                let circleRadius = Math.abs(bts.rxLev) * 0.45;

                mapHTML += `
                    <div class="bts-circle ${isServingBTS ? 'active-circle' : ''}" style="top: ${topPos}%; left: ${leftPos}%; width: ${circleRadius*2}px; height: ${circleRadius*2}px;"></div>
                    <div class="bts-marker ${isServingBTS ? 'active-bts' : ''}" style="top: ${topPos}%; left: ${leftPos}%;"></div>
                    <div class="bts-label ${isServingBTS ? 'active-bts-txt' : ''}" style="top: ${topPos}%; left: ${leftPos}%; font-size:7px;">${bts.cellId}</div>
                `;

                if (isServingBTS) {
                    mapHTML += `
                        <svg class="absolute inset-0 w-full h-full pointer-events-none" style="z-index: 10;">
                            <line x1="${leftPos}%" y1="${topPos}%" x2="50%" y2="50%" stroke="#007f00" stroke-width="1" stroke-dasharray="2,2"/>
                        </svg>
                    `;
                }
            });

            mapHTML += `</div>`;
            return mapHTML;
        }

                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5"><span>Consulter</span><span>${currentInput.length > 0 ? "Effacer" : "Retour"}</span></div>

                    <div class="flex justify-between text-xs border-t border-[#4e5b43] pt-0.5"><span>Confirmer</span><span>${currentInput.length > 0 ? "Effacer" : "Annuler"}</span></div>

        // --- GESTION DU CLAVIER NUMÉRIQUE ---
        function handleKey(key) {
            if (currentMenuState === "REC_NOM" || currentMenuState === "REC_PNOM") {
                if (tapTimer) clearTimeout(tapTimer);
                
                if (lastTapKey === key) {
                    tapIndex++;
                    currentInput = currentInput.slice(0, -1);
                } else {
                    lastTapKey = key;
                    tapIndex = 0;
                }
                
                const chars = multiTapMap[key] || [key];
                currentInput += chars[tapIndex % chars.length];
                
                tapTimer = setTimeout(() => {
                    lastTapKey = null;
                    tapIndex = 0;
                }, 800);
            }
            else if (currentMenuState === "MAIN" || currentMenuState === "MENU_USSD" || currentMenuState === "RECOLE_NUM" || currentMenuState === "RECOLE_MONTANT") {
                currentInput += key;
            }
            else if (currentMenuState === "RECOLE_PIN" || currentMenuState === "TRANS_PIN") {
                if (currentInput.length < 4 && !isNaN(key)) currentInput += key;
            }
            else if (currentMenuState === "REC_DOB") {
                if (currentInput.length < 8 && !isNaN(key)) currentInput += key;
            }
            else if (["SOLDE", "ERREUR", "TRANS_OK"].includes(currentMenuState)) {
                if (!isNaN(key)) {
                    currentInput = key;
                    currentMenuState = "MAIN";
                }
            }
            renderScreen();
        }

        function handleClear() {
            if (tapTimer) { clearTimeout(tapTimer); lastTapKey = null; tapIndex = 0; }
            if (currentInput.length > 0) {
                currentInput = currentInput.slice(0, -1);
                renderScreen();
            }
        }

        // --- GESTION DU D-PAD ---
        function handleArrowUp() { 
            if (currentMenuState === "MENU_USSD" && menuIndex > 0) { menuIndex--; renderScreen(); } 
            else if (currentMenuState === "MENU_APP" && menuIndex > 0) { menuIndex--; renderScreen(); }
            else if (currentMenuState === "APP_LOC_RESULT" && locViewIndex === 0 && btsListScroll > 0) { 
                btsListScroll--; renderScreen();
            }
        }
        function handleArrowDown() { 
            if (currentMenuState === "MENU_USSD" && menuIndex < ussdOptions.length - 1) { menuIndex++; renderScreen(); } 
            else if (currentMenuState === "MENU_APP" && menuIndex < appOptions.length - 1) { menuIndex++; renderScreen(); }
            else if (currentMenuState === "APP_LOC_RESULT" && locViewIndex === 0 && btsListScroll < 2) { 
                btsListScroll++; renderScreen();
            }
        }
        function handleArrowLeft() {
             if (currentMenuState === "APP_LOC_RESULT") {
                 locViewIndex--;
                 if (locViewIndex < 0) locViewIndex = MAX_LOC_VIEWS - 1;
                 renderScreen();
             }
        }
        function handleArrowRight() {
            if (currentMenuState === "APP_LOC_RESULT") {
                 locViewIndex++;
                 if (locViewIndex >= MAX_LOC_VIEWS) locViewIndex = 0;
                 renderScreen();
             }
        }
        
        function handleCenterButton() {
            if (currentMenuState === "MENU_APP" || currentMenuState === "MENU_USSD" || currentMenuState === "MAIN") {
                handleSoftLeft(); 
            }
            else if (currentMenuState === "APP_LOC_RESULT") {
                simulationMode = simulationMode === 1 ? 2 : 1;
                simProgress = 0.0; 
                mettreAJourPositionEtSignaux();
            }
        }

        let scanTimeoutId1 = null;

        // --- BOUTON PHYSIQUE GAUCHE (Soft-Left) ---
        function handleSoftLeft() {
            if (tapTimer) { clearTimeout(tapTimer); lastTapKey = null; tapIndex = 0; }
            
            if (currentMenuState === "MENU_APP") {
                if (menuIndex === 4) { // Localisation
                    currentMenuState = "APP_LOC_SCAN";
                    simulationMode = 1; 
                    renderScreen();
                    
                    // 1. D'abord on calcule la position locale
                    const locData = simulerCalculCellID(simulationMode);
                    const top3 = locData.bts.slice(0,3);
                    // On formate les données des BTS (ex: 10243:-60,20184:-70,30948:-75)
                    const btsStr = top3.map(b => `${b.cellId}:${b.rxLev}`).join(',');

                    // 2. On envoie au HLR pour validation via ESP32
                    scanTimeoutId1 = setTimeout(() => {
                        envoyerCommandeESP32("LOC", btsStr, "0", "0", ""); 
                    }, 1500);

                } else {
                    backendMessage = "Application non installee sur cette ROM.";
                    currentMenuState = "ERREUR";
                    renderScreen();
                }
            }
            else if (currentMenuState === "MENU_USSD") {
                let choix = currentInput !== "" ? currentInput : (menuIndex + 1).toString();
                if (choix === "1") { currentInput = ""; currentMenuState = "RECOLE_PIN"; } 
                else if (choix === "2") { transactionOption = "Envoyer de l'argent"; currentInput = ""; currentMenuState = "RECOLE_NUM"; } 
                else { backendMessage = "Option non supportee"; currentMenuState = "ERREUR"; }
            } 
            else if (currentMenuState === "RECOLE_NUM") {
                if (currentInput.length >= 8) { 
                    transactionNumber = currentInput; 
                    currentInput = ""; 
                    currentMenuState = "RECOLE_MONTANT"; 
                }
            }
            else if (currentMenuState === "RECOLE_MONTANT") {
                if (currentInput.length > 0 && parseFloat(currentInput) > 0) { 
                    transactionAmount = currentInput; 
                    currentInput = ""; 
                    currentMenuState = "TRANS_PIN"; 
                }
            }
            else if (currentMenuState === "RECOLE_PIN") {
                if (currentInput.length === 4) {
                    transactionPIN = currentInput; 
                    currentMenuState = "CHARGEMENT"; 
                    renderScreen();
                    setTimeout(() => { envoyerCommandeESP32("SOLDE", "Consultation", "Aucun", "0", transactionPIN); }, 800);
                }
            }
            else if (currentMenuState === "TRANS_PIN") {
                if (currentInput.length === 4) {
                    transactionPIN = currentInput; 
                    currentMenuState = "CHARGEMENT"; 
                    renderScreen();
                    setTimeout(() => { envoyerCommandeESP32("CONFIRME", transactionOption, transactionNumber, transactionAmount, transactionPIN); }, 800);
                }
            }
            else if (currentMenuState === "REC_NOM") {
                recNom = currentInput; currentInput = ""; currentMenuState = "REC_PNOM";
            }
            else if (currentMenuState === "REC_PNOM") {
                recPnom = currentInput; currentInput = ""; currentMenuState = "REC_DOB";
            }
            else if (currentMenuState === "REC_DOB") {
                if(currentInput.length === 8) {
                    currentMenuState = "CHARGEMENT"; renderScreen();
                    setTimeout(() => {
                        envoyerCommandeESP32("CONFIRME", "RECOVERY_" + recNom, "000", recPnom, currentInput);
                    }, 800);
                }
            }
            else if (currentMenuState === "APP_LOC_RESULT") {
                locViewIndex = (locViewIndex + 1) % MAX_LOC_VIEWS;
                renderScreen();
            }
            else if (["SOLDE", "ERREUR", "TRANS_OK"].includes(currentMenuState)) {
                currentInput = "";
                currentMenuState = "MAIN";
            }
            renderScreen();
        }

        // --- BOUTON PHYSIQUE DROITE (Soft-Right) ---
        function handleSoftRight() {
            if (tapTimer) { clearTimeout(tapTimer); lastTapKey = null; tapIndex = 0; }
            
            const textEntryStates = ["MAIN", "MENU_USSD", "RECOLE_NUM", "RECOLE_MONTANT", "RECOLE_PIN", "TRANS_PIN", "REC_NOM", "REC_PNOM", "REC_DOB"];
            if (currentInput.length > 0 && textEntryStates.includes(currentMenuState)) {
                handleClear();
                return;
            }
            
            if (currentMenuState === "MAIN") {
                currentMenuState = "MENU_APP";
                menuIndex = 0;
            }
            else if (currentMenuState === "MENU_APP") {
                currentMenuState = "MAIN";
            }
            else if (currentMenuState === "APP_LOC_SCAN" || currentMenuState === "APP_LOC_RESULT") {
                if(scanTimeoutId1) clearTimeout(scanTimeoutId1);
                arreterLoopSimulation(); 
                currentMenuState = "MENU_APP";
            }
            else if (currentMenuState === "MENU_USSD") { 
                currentMenuState = "MAIN"; 
            } 
            else if (currentMenuState === "RECOLE_NUM") { 
                currentMenuState = "MENU_USSD"; 
            }
            else if (currentMenuState === "RECOLE_MONTANT") { 
                currentMenuState = "RECOLE_NUM"; 
                currentInput = transactionNumber; 
            }
            else if (currentMenuState === "RECOLE_PIN") { 
                currentMenuState = "MENU_USSD"; currentInput = ""; 
            }
            else if (currentMenuState === "TRANS_PIN") { 
                currentMenuState = "RECOLE_MONTANT"; 
                currentInput = transactionAmount; 
            }
            else if (["REC_NOM", "REC_PNOM", "REC_DOB"].includes(currentMenuState)) {
                currentMenuState = "MAIN";
            }
            else if (["SOLDE", "ERREUR", "TRANS_OK"].includes(currentMenuState)) {
                currentMenuState = "MAIN";
            }
            
            renderScreen();
        }

        // --- APPEL VERT ---
        function handleCall() {
            if (currentMenuState === "MAIN") {
                if (currentInput === "*1234#") {
                    currentMenuState = "CHARGEMENT"; 
                    renderScreen();
                    setTimeout(() => { 
                        envoyerCommandeESP32("APPEL_USSD", "*1234#", "Aucun", "0"); 
                    }, 800);
                } else if (currentInput.length > 0) {
                    backendMessage = "Code MMI non valide."; 
                    currentMenuState = "ERREUR"; 
                    renderScreen();
                }
            } else { 
                handleSoftLeft(); 
            }
        }

        function handleEnd() { 
            if(scanTimeoutId1) clearTimeout(scanTimeoutId1);
            arreterLoopSimulation();
            currentInput = ""; 
            currentMenuState = "MAIN"; 
            renderScreen(); 
        }

        renderScreen();

        document.addEventListener('keydown', (e) => {
            if ((e.key >= '0' && e.key <= '9') || e.key === '*' || e.key === '#') {
                handleKey(e.key);
            } 
            else if ((e.key >= 'a' && e.key <= 'z') || (e.key >= 'A' && e.key <= 'Z') || e.key === ' ') {
                if (currentMenuState === "REC_NOM" || currentMenuState === "REC_PNOM") {
                    currentInput += e.key.toUpperCase();
                    renderScreen();
                }
            }
            else if (e.key === 'Backspace') { handleClear(); } 
            else if (e.key === 'Enter') { handleCall(); } 
            else if (e.key === 'Escape') { handleEnd(); } 
            else if (e.key === 'ArrowUp') { handleArrowUp(); } 
            else if (e.key === 'ArrowDown') { handleArrowDown(); }
            else if (e.key === 'ArrowLeft') { handleArrowLeft(); } 
            else if (e.key === 'ArrowRight') { handleArrowRight(); }
            else if (e.key === 'c' || e.key === 'C') { handleCenterButton(); }
        });
